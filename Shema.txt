from DataB import DataPath
import sqlite3
class Tutor:
	def __init__(self,ID):
		"""Tutor Table Contains columns: Company of type integer|| TutorGroup of type integer|| TutorTeacher of type text|| Foreign Keys: Tutor.Company refrences Company.ID"""
		conn = sqlite3.connect(DataPath)
		c= conn.cursor()
		c.execute('''SELECT Company FROM Tutor WHERE ID=?''',ID)
		self.Company=c.fetchall()
		c.execute('''SELECT TutorGroup FROM Tutor WHERE ID=?''',ID)
		self.TutorGroup=c.fetchall()
		c.execute('''SELECT TutorTeacher FROM Tutor WHERE ID=?''',ID)
		self.TutorTeacher=c.fetchall()
